Ejercicio 10

Contar las vocales

            
              const vocales = ["a", "e", "i", "o", "u"]
              
              let string = prompt('Dite algo ahi ');
              let resultado = contarvocales(string);
              document.getElementById('add_after_me10').innerText+=resultado + " vocales en el texto introducido";
              function contarvocales(str) {
                let contar = 0;
                for (let letra of str.toLowerCase()) {
                    if (vocales.includes(letra)) {
                        contar++;
                    }
                }
                return contar
               }
